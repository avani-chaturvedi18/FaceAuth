23 {

25  {
26  {0x02U, 0x01U, 0x01U, 0x03U},
27  {0x03U, 0x02U, 0x01U, 0x01U},
28  {0x01U, 0x03U, 0x02U, 0x01U},
29  {0x01U, 0x01U, 0x03U, 0x02U}
30  };

32  GF256 gf256(0xf5);

34  for (int i = 0; i < 4; i++)

38  for (unsigned int k = 0; k < 4; k++)
