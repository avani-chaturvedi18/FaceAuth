51 void Salsa20_Core(word32* data, unsigned int rounds)
52 {
53  CRYPTOPP_ASSERT(data != NULLPTR);
54  CRYPTOPP_ASSERT(rounds % 2 == 0);
55 
56  CRYPTOPP_ALIGN_DATA(ALIGN_SPEC) word32 x[16];
57 
58  for (size_t i = 0; i < 16; ++i)
59  x[i] = data[i];
60 
61  // Rounds must be even
62  for (size_t i = 0; i < rounds; i += 2)
63  {
64  x[ 4] ^= rotlConstant< 7>(x[ 0]+x[12]);
65  x[ 8] ^= rotlConstant< 9>(x[ 4]+x[ 0]);
66  x[12] ^= rotlConstant<13>(x[ 8]+x[ 4]);
67  x[ 0] ^= rotlConstant<18>(x[12]+x[ 8]);
68 
69  x[ 9] ^= rotlConstant< 7>(x[ 5]+x[ 1]);
70  x[13] ^= rotlConstant< 9>(x[ 9]+x[ 5]);
71  x[ 1] ^= rotlConstant<13>(x[13]+x[ 9]);
72  x[ 5] ^= rotlConstant<18>(x[ 1]+x[13]);
73 
74  x[14] ^= rotlConstant< 7>(x[10]+x[ 6]);
75  x[ 2] ^= rotlConstant< 9>(x[14]+x[10]);
76  x[ 6] ^= rotlConstant<13>(x[ 2]+x[14]);
77  x[10] ^= rotlConstant<18>(x[ 6]+x[ 2]);
78 
79  x[ 3] ^= rotlConstant< 7>(x[15]+x[11]);
80  x[ 7] ^= rotlConstant< 9>(x[ 3]+x[15]);
81  x[11] ^= rotlConstant<13>(x[ 7]+x[ 3]);
82  x[15] ^= rotlConstant<18>(x[11]+x[ 7]);
83 
84  x[ 1] ^= rotlConstant< 7>(x[ 0]+x[ 3]);
85  x[ 2] ^= rotlConstant< 9>(x[ 1]+x[ 0]);
86  x[ 3] ^= rotlConstant<13>(x[ 2]+x[ 1]);
87  x[ 0] ^= rotlConstant<18>(x[ 3]+x[ 2]);
88 
89  x[ 6] ^= rotlConstant< 7>(x[ 5]+x[ 4]);
90  x[ 7] ^= rotlConstant< 9>(x[ 6]+x[ 5]);
91  x[ 4] ^= rotlConstant<13>(x[ 7]+x[ 6]);
92  x[ 5] ^= rotlConstant<18>(x[ 4]+x[ 7]);
93 
94  x[11] ^= rotlConstant< 7>(x[10]+x[ 9]);
95  x[ 8] ^= rotlConstant< 9>(x[11]+x[10]);
96  x[ 9] ^= rotlConstant<13>(x[ 8]+x[11]);
97  x[10] ^= rotlConstant<18>(x[ 9]+x[ 8]);
98 
99  x[12] ^= rotlConstant< 7>(x[15]+x[14]);
100  x[13] ^= rotlConstant< 9>(x[12]+x[15]);
101  x[14] ^= rotlConstant<13>(x[13]+x[12]);
102  x[15] ^= rotlConstant<18>(x[14]+x[13]);
103  }
104 
105 // OpenMP 4.0 released July 2013.
106 #if _OPENMP >= 201307
107  #pragma omp simd
108  for (size_t i = 0; i < 16; ++i)
109  data[i] += x[i];
110 #else
111  for (size_t i = 0; i < 16; ++i)
112  data[i] += x[i];
113 #endif
114 }

125 void Salsa20_Policy::CipherSetKey(const NameValuePairs &params, const byte *key, size_t length)
126 {
127  // Use previous rounds as the default value
128  int rounds = params.GetIntValueWithDefault(Name::Rounds(), m_rounds);
129  if (rounds != 20 && rounds != 12 && rounds != 8)
130  throw InvalidRounds(Salsa20::StaticAlgorithmName(), rounds);
131 
132  // Latch a good value
133  m_rounds = rounds;
134 
135  // m_state is reordered for SSE2
136  GetBlock<word32, LittleEndian> get1(key);
137  get1(m_state[13])(m_state[10])(m_state[7])(m_state[4]);
138  GetBlock<word32, LittleEndian> get2(key + length - 16);
139  get2(m_state[15])(m_state[12])(m_state[9])(m_state[6]);
140 
141  // "expand 16-byte k" or "expand 32-byte k"
142  m_state[0] = 0x61707865;
143  m_state[1] = (length == 16) ? 0x3120646e : 0x3320646e;
144  m_state[2] = (length == 16) ? 0x79622d36 : 0x79622d32;
145  m_state[3] = 0x6b206574;
146 }

149 {
150  CRYPTOPP_UNUSED(keystreamBuffer), CRYPTOPP_UNUSED(length);

153  GetBlock<word32, LittleEndian> get(IV);
154  get(m_state[14])(m_state[11]);
155  m_state[8] = m_state[5] = 0;
