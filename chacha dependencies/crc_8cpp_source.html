147 void CRC32::Update(const byte *s, size_t n)
148 {
149 #if (CRYPTOPP_ARM_CRC32_AVAILABLE)
150  if (HasCRC32())
151  {
152  CRC32_Update_ARMV8(s, n, m_crc);
153  return;
154  }
155 #endif
156 
157  word32 crc = m_crc;
158 
159  for(; !IsAligned<word32>(s) && n > 0; n--)
160  crc = m_tab[CRC32_INDEX(crc) ^ *s++] ^ CRC32_SHIFTED(crc);
161 
162  while (n >= 4)
163  {
164  crc ^= *(const word32 *)(void*)s;
165  crc = m_tab[CRC32_INDEX(crc)] ^ CRC32_SHIFTED(crc);
166  crc = m_tab[CRC32_INDEX(crc)] ^ CRC32_SHIFTED(crc);
167  crc = m_tab[CRC32_INDEX(crc)] ^ CRC32_SHIFTED(crc);
168  crc = m_tab[CRC32_INDEX(crc)] ^ CRC32_SHIFTED(crc);
169  n -= 4; s += 4;
170  }
171 
172  while (n--)
173  crc = m_tab[CRC32_INDEX(crc) ^ *s++] ^ CRC32_SHIFTED(crc);
174 
175  m_crc = crc;
176 }

178 void CRC32::TruncatedFinal(byte *hash, size_t size)
179 {
180  ThrowIfInvalidTruncatedSize(size);
181 
182  m_crc ^= CRC32_NEGL;
183  for (size_t i=0; i<size; i++)
184  hash[i] = GetCrcByte(i);
185 
186  Reset();
187 }
