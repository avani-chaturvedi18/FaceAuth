13 // RIPEMD-160 definitions used by Two-Track-MAC
14 word32 F(word32 x, word32 y, word32 z) { return x ^ y ^ z; }
15 word32 G(word32 x, word32 y, word32 z) { return z ^ (x & (y^z)); }
16 word32 H(word32 x, word32 y, word32 z) { return z ^ (x | ~y); }
17 word32 I(word32 x, word32 y, word32 z) { return y ^ (z & (x^y)); }
18 word32 J(word32 x, word32 y, word32 z) { return x ^ (y | ~z); }
