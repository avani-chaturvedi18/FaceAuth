96  if (!mbi.AllocationBase || mbi.AllocationBase == hModule)
97  continue;

100  PGetNewAndDelete pGetNewAndDelete = (PGetNewAndDelete)GetProcAddress(hModule, "GetNewAndDeleteForCryptoPP");
101  if (pGetNewAndDelete)
